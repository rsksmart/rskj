// RSKIP-545 deep DSL scenarios — spec sections not covered by rskip545Test.txt:
//   - Multiple tuples for same authority: last valid occurrence wins (Behavior)
//   - Invalid tuple skipped, next tuple processed (Behavior step 1-8)
//   - Dual authority in one transaction (sponsored batch pattern)
//   - Delegate to precompile address (No Precompiles section)
//   - Value transfer credited after authorization phase
//   - Calldata intrinsic gas on Type 4 payload
//   - Invalid tuple still incurs PER_EMPTY_ACCOUNT_COST upfront (Gas Costs)
//   - Delegation persists when delegated execution reverts (Behavior: not rolled back)
//   - Triple authorization tuple upfront gas (Gas Costs)
//   - Universal chain ID (0) accepted on authorization tuple
//   - Duplicate authorization tuples — first applied, gas for both
//   - Invalid-first ordering — second valid tuple still applied

end

account_new sponsor 10000000000000000000
account_new acc2 0
account_new acc3 10000000000000000000
account_new revertImpl 0 60006000fd

# Delegate targets used across scenarios
#   0x...0003 — regtest stub (same as main suite)
#   0x...0004 — alternate delegate for last-wins test
#   0x...0001 — precompile address (must not execute as delegate code)

# ============================================================================
# b01: Same authority, two valid tuples — last delegate address wins
# sponsor tx nonce 0; auth nonces 1 then 2 (post sender-increment rule)
# ============================================================================

transaction_build txMultiAuthLastWins
    sender sponsor
    receiver acc2
    value 0
    gas 200000
    maxPriorityFeePerGas 1000000000
    maxFeePerGas 2000000000
    transactionType 04
    authorization sponsor 0000000000000000000000000000000000000003 1
    authorization sponsor 0000000000000000000000000000000000000004 2
    build

block_build b01
    parent g00
    transactions txMultiAuthLastWins
    build

block_connect b01

assert_best b01
assert_tx_success txMultiAuthLastWins
