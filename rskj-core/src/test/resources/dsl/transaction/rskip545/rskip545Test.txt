// RSKIP-545 DSL Test: Type 4 (EIP-7702 set-code) transactions in Rootstock
//
// Self-auth nonce rule: sender nonce is incremented before auth processing, so when the
// authority is the sender the auth tuple nonce must equal (account nonce after sender increment).
//
// Revoke leaves CODE_FOR_CLEANING on the authority; that account cannot originate further
// Type 4 transactions until delegation is restored (see Rskip545DslTest).

end

account_new acc1 10000000000000000000
account_new acc2 1000000
account_new authority 10000000000000000000

# Test 1: Self-authorization (tx nonce 0, auth nonce 1)

transaction_build txType4SelfAuth
    sender acc1
    receiver acc2
    value 0
    gas 100000
    maxPriorityFeePerGas 1000000000
    maxFeePerGas 2000000000
    transactionType 04
    authorization acc1 0000000000000000000000000000000000000003 1
    build

block_build b01
    parent g00
    transactions txType4SelfAuth
    build

block_connect b01

assert_best b01
assert_tx_success txType4SelfAuth
