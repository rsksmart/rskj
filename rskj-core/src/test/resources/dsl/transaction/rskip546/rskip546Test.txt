// RSKIP-546 DSL Test: Type 1 and Type 2 Transaction Encoding in Rootstock
//
// Tests the full RSKIP-546 specification:
//   1. Type 1 (EIP-2930) transactions execute correctly
//   2. Standard Type 2 (EIP-1559 format) transactions execute correctly
//   3. Effective gas price = min(maxPriorityFeePerGas, maxFeePerGas) for Type 2
//      - When maxPriority < maxFee, effectiveGasPrice == maxPriorityFeePerGas
//      - When maxPriority == maxFee, effectiveGasPrice == either (they are equal)
//   4. Type 1 receipt encoding: 0x01 || rlp([status, cumulativeGas, bloom, logs])
//   5. Type 2 receipt encoding: 0x02 || rlp([status, cumulativeGas, bloom, logs])
//   6. Access list bytes are accepted but not executed (RSK ignores access list contents)
//   7. Non-empty access list incurs extra gas (80 gas/byte per RSKIP-546)
//   8. Empty access list (0xc0) does NOT incur extra gas
//   9. Mixed block: Type 1, Type 2, and legacy in the same block
//  10. Block encode/decode preserves typed transactions
//  11. Activation guard: Type 2 is rejected before RSKIP-546 is active (separate world)

end

# Create test accounts with sufficient balance
account_new acc1 10000000000000000000
account_new acc2 1000000
account_new acc3 1000000

# ============================================================================
# Test 1: Basic Type 1 transaction (EIP-2930 format, empty access list)
# ============================================================================
# Type 1: 0x01 || rlp([chainId, nonce, gasPrice, gasLimit, to, value, data,
#                       accessList, signatureYParity, signatureR, signatureS])

transaction_build txType1Basic
    sender acc1
    receiver acc2
    value 1000
    gas 21000
    gasPrice 1000000000
    transactionType 01
    build

block_build b01
    parent g00
    transactions txType1Basic
    build

block_connect b01

assert_best b01
assert_tx_success txType1Basic

# ============================================================================
# Test 2: Basic Type 2 transaction (EIP-1559 format, maxPriority < maxFee)
# ============================================================================
# Effective gas price = min(maxPriorityFeePerGas, maxFeePerGas)
#   = min(500000000, 2000000000) = 500000000

transaction_build txType2PriorityLower
    sender acc1
    nonce 1
    receiver acc2
    value 2000
    gas 21000
    maxPriorityFeePerGas 500000000
    maxFeePerGas 2000000000
    transactionType 02
    build

block_build b02
    parent b01
    transactions txType2PriorityLower
    build

block_connect b02

assert_best b02
